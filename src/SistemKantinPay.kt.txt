class SistemKantinPay {

    fun beliMenu(mahasiswa: Mahasiswa, menu: Menu, pin: String) {

        if (!mahasiswa.verifikasiPin(pin)) {
            println("Gagal: PIN salah")
            return
        }

        if (menu.getStok() <= 0) {
            println("Gagal: Stok habis")
            return
        }

        if (!mahasiswa.potongSaldo(menu.harga)) {
            println("Gagal: Saldo tidak cukup")
            return
        }

        menu.kurangiStok()
        println("Berhasil membeli ${menu.nama}")
    }
}
